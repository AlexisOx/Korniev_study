// Перевірка версії Git
git -v

// Відкрити файл налаштувань Git
git config --global --edit

// Встановлення ім'я користувача та емейл
git config --global user.name "Your Name"
git config --global user.email "your_email@whatever.com"

// Міняти crlf на lf
git config --global core.autocrlf input

// Згенерити новий ssh-ключ
ssh-keygen -o -t rsa -C "your_email@whatever.com"

// Перейти в папку з ключем (Для автозавершення команди можна використати Tab)
cd ~/.ssh/

//  Відкрити файл
cat id_rsa.pub

// Перевірка автентифікації в gitbush ssh key
$ ssh -T git@github.com

// Подивитись, які файли знаходяться в папці
ls
//  Подивитись, які файли знаходяться в папці, включаючи приховані
ls -a

// Скопіювати репозиторій
git clone ссылка

// Перейти в папку
cd "Назва папки"

// Повернутись на початкову папку
pwd

// Додати файл
touch "Назва файлу.тип файлу"

// Відкрити редактор
nano "Файл" (Також можна сторити файл цією командою)
Вийти з редактора Ctrl+X
Зберегти зміни Shift+Y
Entr

//Переглянути що містить файл
Cat "Файл"

// Додати файл в індекс
git add ім'я файла
git add . (Додає всі файли)

//Видалити всі зміни "Видалить все тільки якщо добавлено в індекс"
git reset --hard

// Зберегти зміни (закоммітити)
git commit -m 'Коментар'

// Зберегти в GitHub (запушити)
git push

// Викачати останні зміни з репозиторію (Якщо хтось щось зробив, і треба зберегти собі на комп)
git pull

// Створити гілку 
git checkout -b "Назва гілки"

// Перейти між гілками
git checkout "master"

//Зберегти зміни саме в створеній гілці
git push --set-upstream origin Назва гілки
git checkout "name.surname.gitpractice"
